Program 11: Java Script Program to create application form


<!DOCTYPE html>
<html>
<head>
  <title>Application Form</title>
  <style>
    body { font-family: Arial, sans-serif; padding: 20px; }
    input, textarea { display: block; margin: 10px 0; width: 300px; }
    button { padding: 5px 10px; }
  </style>
</head>

<body>
  <h2>Application Form</h2>

  <form id="appForm">
    <input type="text" id="name" placeholder="Full Name" required>
    <input type="email" id="email" placeholder="Email" required>
    <input type="number" id="age" placeholder="Age" required>
    <textarea id="address" placeholder="Address" required></textarea>
    <button type="submit">Submit</button>
  </form>

  <p id="result"></p>

  <script>
    document.getElementById("appForm").addEventListener("submit", function(e) {
      e.preventDefault();

      const name = document.getElementById("name").value;
      const email = document.getElementById("email").value;
      const age = document.getElementById("age").value;
      const address = document.getElementById("address").value;

      document.getElementById("result").innerText =
        `Submitted:\nName: ${name}\nEmail: ${email}\nAge: ${age}\nAddress: ${address}`;

      this.reset();
    });
  </script>
</body>
</html>

Program 14: Java Script Program to find factorial using external JS


<!DOCTYPE html>
<html>
<head>
  <title>Factorial Calculator</title>
  <script src="factorial.js"></script>
</head>
<body>
  <h2>Factorial Finder</h2>
  <label>Enter a number:</label>
  <input type="number" id="number" />
  <button onclick="calculateFactorial()">Calculate Factorial</button>
  <p id="result"></p>
</body>
</html>


function calculateFactorial()
{
  var num = parseInt(document.getElementById("number").value);
  var resultElement = document.getElementById("result");

  if (isNaN(num) || num < 0) {
    resultElement.innerHTML = "Please enter a non-negative integer.";
    return;
  }

  var factorial = 1;
  for (var i = 1; i <= num; i++) {
    factorial *= i;
  }

  resultElement.innerHTML =
    "Factorial of " + num + " is: " + factorial;
}

Program 15: Java Script Program to print Fibonacci series


<!DOCTYPE html>
<html>
<head>
  <title>Fibonacci Series</title>
</head>
<body>
  <h2>Fibonacci Series Generator</h2>
  <label>Enter number of terms (N):</label>
  <input type="number" id="terms" />
  <button onclick="generateFibonacci()">Generate</button>
  <p id="result"></p>

  <script>
   
function generateFibonacci() {
  let n = parseInt(document.getElementById("terms").value);
  let resultElement = document.getElementById("result");

  if (isNaN(n) || n <= 0) {
    resultElement.innerHTML = "Please enter a positive integer.";
    return;
  }

  let fibSeries = [];
  let a = 0, b = 1;

  for (let i = 0; i < n; i++) {
    fibSeries.push(a);
    let temp = a + b;
    a = b;
    b = temp;
  }

  resultElement.innerHTML =
    "Fibonacci Series up to " + n + " terms
" +
    fibSeries.join(", ");
}


  </script>
</body>
</html>
